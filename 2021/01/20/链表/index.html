1.移除链表元素

法一、双指针

struct ListNode* removeElements(struct ListNode* head, int val){
struct ListNode *p,*q;//p为当前结点，q为p前面一个节点
while(head!=NULL&&head->val==val){
    head=head->next;
}
if(head==NULL)return NULL;//防止整个链表都是应删除元素的情况和空表的情况
p=head->next;
q=head;//初始化p，q
while(p!=NULL){
    if(p->val==val)q->next=p->next;//如果要删除，q指向p的下一个结点，p结点被删除
else q=p;//q向后移动
p=p->next;//p向后遍历
}
return head;
}

法二、递归

struct ListNode* removeElements(struct ListNode* head, int val){
    if(head==NULL)return NULL;
    head->next=removeElements(head->next,val);
    return head->val==val?head->next:head;
}

如果头节点的数据和val相同，则返回去头节点的链表；如果不同，就返回原链表。以此递归，则可删除所有与val相同的元素。

法三、迭代

struct ListNode* removeElements(struct ListNode* head, int val){
    while (head != NULL && head->val == val) {
        head = head->next;
    }
    if (head == NULL) {
        return NULL;
    }
    struct ListNode* pre = head;
    while (pre->next != NULL) {
        if (pre->next->val == val) {
            pre->next = pre->next->next;//如果相同，pre指向后后个结点
        } else {
            pre = pre->next;//pre向后移动
        }
    }
    return head;
}

2.设计链表

设计链表的实现。您可以选择使用单链表或双链表。单链表中的节点应该具有两个属性：val 和 next。val 是当前节点的值，next 是指向下一个节点的指针/引用。如果要使用双向链表，则还需要一个属性 prev 以指示链表中的上一个节点。假设链表中的所有节点都是 0-index 的。

在链表类中实现这些功能：

get(index)：获取链表中第 index 个节点的值。如果索引无效，则返回-1。
addAtHead(val)：在链表的第一个元素之前添加一个值为 val 的节点。插入后，新节点将成为链表的第一个节点。
addAtTail(val)：将值为 val 的节点追加到链表的最后一个元素。
addAtIndex(index,val)：在链表中的第 index 个节点之前添加值为 val  的节点。如果 index 等于链表的长度，则该节点将附加到链表的末尾。如果 index 大于链表长度，则不会插入节点。如果index小于0，则在头部插入节点。
deleteAtIndex(index)：如果索引 index 有效，则删除链表中的第 index 个节点。
示例:

MyLinkedList linkedList = new MyLinkedList();
linkedList.addAtHead(1);
linkedList.addAtTail(3);
linkedList.addAtIndex(1,2);   // 链表变为1-> 2-> 3
linkedList.get(1);            // 返回2
linkedList.deleteAtIndex(1);  // 现在链表是1-> 3
linkedList.get(1);            // 返回3

提示：

所有val值都在 [1, 1000] 之内。
操作次数将在 [1, 1000] 之内。
请不要使用内置的 LinkedList 库。

1、创建结构体

typedef struct Node{
    int val;//结点数据
    struct Node *next;//结点指针
} MyLinkedList;//定义结构体名字

2、创建链表

创建一个虚无又有用的头结点

MyLinkedList* myLinkedListCreate() {
    MyLinkedList *head;
    head = calloc(1, sizeof(MyLinkedList));// 为头结点开辟存储空间
    if (!head)
        return NULL;// 头结点为空的情况
    head->val = -1;// 头结点取值随意
    head->next = NULL;
    return head;// 返回头结点
}

3、获取链表中第index个结点的值

int myLinkedListGet(MyLinkedList* obj, int index) {
    if (!obj || index < 0)return -1;/*index小于0，索引无效，返回-1*/
    while (obj && index >= 0) {
        obj = obj->next;
        index--;
    }//遍历搜索第index个结点
    if (!obj)return -1;/*如果搜索到的结点为空，索引无效，返回-1*/
    return obj->val;/*若有效，返回搜索到的结点的值*/
}

4、在第一个元素前添加结点

void myLinkedListAddAtHead(MyLinkedList* obj, int val) {
    MyLinkedList *new = calloc(1, sizeof(MyLinkedList));// 为新结点开辟存储空间
    if (!new)return;// 若新节点为空，返回
    new->next = obj->next;
    new->val = val;
    obj->next = new;// 将新结点插在虚无表头后，第一个结点前，称为第一个结点
}

5、在最后一个元素后添加结点

void myLinkedListAddAtTail(MyLinkedList* obj, int val) {
    MyLinkedList *new;
    new = calloc(1, sizeof(MyLinkedList));// 为新结点开辟存储空间
    while (obj->next)
    {
        obj = obj->next;
    }// 遍历到最后一个结点
    new->val = val;
    new->next = NULL;
    obj->next = new;
}// 将新节点插在最后一个结点后面，并指向空

6、在第index个结点前添加结点

void myLinkedListAddAtIndex(MyLinkedList* obj, int index, int val) {
    MyLinkedList *new;
    if (index < 0)return;
    new = calloc(1, sizeof(MyLinkedList));
    index--;
    while (obj && index >= 0)
    {
        obj = obj->next;
        index--;/*遍历到第index-1个结点，obj为原来的第index之前一个结点*/
    }
    if (!obj)
        return;
    new->val = val;
    new->next = obj->next;/*new 结点指向第index个结点，称为第index个结点之前的结点*/
    obj->next = new;/*原来index之前的结点指向new结点*/
}/*因为虚无表头的存在，头尾同样适用*/

7、删除链表中的第index个节点

void myLinkedListDeleteAtIndex(MyLinkedList* obj, int index) {
    MyLinkedList *tmp;
    index--;
    while (obj && index >= 0)
    {
        obj = obj->next;
        index--;/*遍历到第index-1个结点*/
    }
    if (!obj)
        return;/*若第index-1个元素为空，返回*/
    tmp = obj->next;
    if (!tmp)
        return;/*若第index个元素为空，返回*/
    if (obj->next->next)
        obj->next = obj->next->next;/*若第index+1个元素不为空，第index-1个结点指向第index+1个结点*/
    else
        obj->next = NULL;/*若第index+1个元素为空，第index-1个结点指向空*/
    free(tmp);/*释放原来第index个结点的内存*/
}

8、释放链表

void myLinkedListFree(MyLinkedList* obj) {
    MyLinkedList *tmp;
    while (obj) {
        tmp = obj->next;
        free(obj);
        obj = tmp;
    }
}

3.反转链表

struct ListNode* reverseList(struct ListNode* head) {
    struct ListNode* q = NULL;
    struct ListNode* p = head;
    while (p) {
        struct ListNode* t = p->next;/*向后移动*/
        p->next = q;/*反转链表*/
        q = p;/*反转链表*/
        p = t;/*向后移动*/
    }
    return q;
}

4.快慢指针

思路与算法：

我们使用两个指针，fast与slow。它们起始都位于链表的头部。随后，slow指针每次向后移动一个位置，而fast指针向后移动两个位置。如果链表中存在环，则fast指针最终将再次与slow指针在环中相遇。
如下图所示，设链表中环外部分的长度为a。slow指针进入环后，又走了b的距离与fast相遇。此时，fast指针已经走完了环的n圈，因此它走过的总距离为a+n(b+c)+b=a+(n+1)b+nc。

根据题意，任意时刻，fast指针走过的距离都为slow指针的2倍。因此，我们有
a+(n+1)b+nc=2(a+b)⟹a=c+(n−1)(b+c)
有了a=c+(n-1)(b+c)的等量关系，我们会发现：从相遇点到入环点的距离加上n-1圈的环长，恰好等于从链表头部到入环点的距离。
因此，当发现slow与fast相遇时，我们再额外使用一个指针ptr。起始，它指向链表头部；随后，它和slow每次向后移动一个位置。最终，它们会在入环点相遇。

代码：

struct ListNode* detectCycle(struct ListNode* head) {
    struct ListNode *slow = head, *fast = head;
    while (fast != NULL) {
        slow = slow->next;
        if (fast->next == NULL) {
            return NULL;
        }
        fast = fast->next->next;
        if (fast == slow) {
            struct ListNode* ptr = head;
            while (ptr != slow) {
                ptr = ptr->next;
                slow = slow->next;
            }
            return ptr;
        }
    }
    return NULL;
}
